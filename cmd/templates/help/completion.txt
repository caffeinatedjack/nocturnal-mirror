Generate shell completion script for nocturnal.

To load completions:

Bash:
  $ source <(nocturnal completion bash)

  # To load completions for each session, execute once:
  # Linux:
  $ nocturnal completion bash > /etc/bash_completion.d/nocturnal
  # macOS:
  $ nocturnal completion bash > $(brew --prefix)/etc/bash_completion.d/nocturnal

Zsh:
  # If shell completion is not already enabled in your environment,
  # you will need to enable it. You can execute the following once:
  $ echo "autoload -U compinit; compinit" >> ~/.zshrc

  # To load completions for each session, execute once:
  $ nocturnal completion zsh > "${fpath[1]}/_nocturnal"

  # You will need to start a new shell for this setup to take effect.

Fish:
  $ nocturnal completion fish | source

  # To load completions for each session, execute once:
  $ nocturnal completion fish > ~/.config/fish/completions/nocturnal.fish

PowerShell:
  PS> nocturnal completion powershell | Out-String | Invoke-Expression

  # To load completions for every new session, run:
  PS> nocturnal completion powershell > nocturnal.ps1
  # and source this file from your PowerShell profile.
